Redis应用场景 ​

一、缓存 ​

Redis是基于内存的，所以效率比MySQL快1000倍；如果使用Redis做缓存，接口请求先查找Redis，如果找不到再查数据库，能极大提高并发

模拟使用：具体业务需要考虑缓存穿透的问题

py
import redis

# 连接 Redis
r = redis.Redis(host='localhost', port=6379, db=0)

def get_data_from_cache(key):
    """从缓存中获取数据"""
    data = r.get(key)
    if data:
        return data.decode('utf-8')
    return None

def set_data_in_cache(key, value, expiration=3600):
    """将数据存入缓存，并设置过期时间"""
    r.set(key, value, ex=expiration)

def get_data(key):
    """先从缓存获取数据，若缓存中不存在则从数据源获取并更新缓存"""
    cached_data = get_data_from_cache(key)
    if cached_data:
        print(f"从缓存中获取数据: {cached_data}")
        return cached_data
    # 模拟从数据源获取数据
    data = "这是从数据源获取的数据"
    set_data_in_cache(key, data)
    print(f"从数据源获取数据并更新缓存: {data}")
    return data

二、排行榜 ​

使用ZSet实现，ZSet内的数据不重复，还有分数控制，天然适合做排行榜

py
import redis

# 连接 Redis
r = redis.Redis(host='localhost', port=6379, db=0)

def add_score(user_id, score):
    """添加用户分数到排行榜"""
    r.zadd('leaderboard', {user_id: score})

def get_top_n(n):
    """获取排行榜前 n 名"""
    top_n = r.zrevrange('leaderboard', 0, n - 1, withscores=True)
    for user_id, score in top_n:
        print(f"用户 ID: {user_id.decode('utf-8')}, 分数: {score}")

# 使用示例
add_score('user1', 100)
add_score('user2', 200)
add_score('user3', 150)
get_top_n(2)

py
from redisbloom.client import Client

# 这里不能用线程池
bloom_filter = Client(host=settings.CACHE_HOST,port=settings.CACHE_PORT)
# 初始化布隆过滤器
bloom_filter.bfCreate('user_ids', 0.01, 1000000)
bloom_filter.bfAdd('user1')
if not bloom_filter.bfExists('user_ids', 'user5'):
    return {"error": "Product not found"}
