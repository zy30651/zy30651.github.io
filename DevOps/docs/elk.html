# ES Deployment配置文件，用于部署ES集群
apiVersion: apps/v1
kind: StatefulSet # 属于NoSQL,有状态服务创建pod，用于部署ES集群
metadata:
  name: elasticsearch-logging # 有状态服务的pod名称
  namespace: kube-logging
  labels:
    k8s-app: elasticsearch-logging
    kubernetes.io/cluster-service: "true"
    addonmanager.kubernetes.io/mode: Reconcile
    srv: srv-elasticsearch
spec:
  # 服务名称,与srv相关联，可以确保使用一下DNS地址访问StatefulSet中的每个pod
  # 访问方式为：<pod-name>.<service-name>.<namespace>.svc.cluster.local
  serviceName: elasticsearch-logging
  replicas: 1 # 副本数量
  selector: # 选择器，用于选择pod
    matchLabels:
      k8s-app: elasticsearch-logging # 和pod template中的标签匹配
  template: # pod模板
    metadata:
      labels:
        k8s-app: elasticsearch-logging # pod的标签
        kubenetes.io/cluster-service: "true"
    spec:
      serviceAccountName: elasticsearch-logging # 指定服务账号
      containers:
        - image: docker.m.daocloud.io/library/elasticsearch:7.9.3 # ES镜像
          name: elasticsearch-logging # 容器名称
          resources: # 资源限制
            requests:
              cpu: 1000m # CPU请求
              memory: 2Gi # 内存请求
            limits:
              cpu: 100m # CPU限制
              memory: 500Mi # 内存限制
          ports:
            - containerPort: 9200 # 容器端口
              name: db # 容器端口名称
              protocol: TCP # 协议类型
            - containerPort: 9300 # 容器端口
              name: transport # 容器端口名称
              protocol: TCP # 协议类型
          volumeMounts:
            - name: elasticsearch-logging # 挂载卷名称
              mountPath: /usr/share/elasticsearch/data # 挂载路径
          env:
            - name: "NAMESPACE" # 环境变量名称
              valueFrom:
                fieldRef:
                  fieldPath: metadata.namespace # 获取pod的命名空间
            - name: ES_JAVA_OPTS # ES的JVM参数
              value: "-Xms512m -Xmx512m" # 设置JVM的内存限制
            - name: discovery.type # ES集群发现类型
              value: "single-node" # 单节点模式
      volumes:
        - name: elasticsearch-logging # 卷名称
          hostPath:
            path: /data/es/ # 宿主机路径
      initContainers: # 初始化容器，用于在主容器启动之前执行一些操作
        - name: eleasticsearch-logging-init
          image: alpine:3.6 # 初始化容器镜像
          command: ["/sbin/sysctl", "-w", "vm.max_map_count=262144"] # 添加mmap计数限制,太低可能内存不足
          securityContext: # 仅运行在指定的容器上，不会影响Volume
            privileged: true # 以特权模式运行
        - name: increase-fd-ulimit
          image: busybox
          imagePullPolicy: IfNotPresent
          command: ["/bin/sh", "-c", "ulimit -n 65536"] # 设置文件描述符限制,修改并发数
          securityContext:
            privileged: true # 以特权模式运行
        - name: elasticsearch-volume-init # es数据卷初始化容器，加上777权限
          image: alpine:3.6
          command: ["/bin/sh", "-c", "chmod 777 /usr/share/elasticsearch/data"]
          volumeMounts:
            - name: elasticsearch-logging # 挂载卷名称
              mountPath: /usr/share/elasticsearch/data # 挂载路径

kind: Service
metadata:
  name: logstash

spec:
  ports:
    - port: 5044 # Logstash beats端口

---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: logstash
  namespace: kube-logging
spec:
  selector:
    matchLabels:
      type: logstash
  template:
    metadata:
      labels:
        type: logstash
        srv: srv-logstash
    spec:
      containers:
        - image: docker.m.daocloud.io/library/logstash:7.9.3
          name: logstash
          ports:
            - containerPort: 5044 # Logstash的beats端口
              name: beats # 容器端口名称
          command:
            - logstash
            - "-f"
            - "/etc/logstash_c/logstash.conf"
          env:
            - name: "XPACK_MONITORING_ELASTICSEARCH_HOSTS" # 设置XPack监控的Elasticsearch主机
              value: "http://elasticsearch-logging:9200"
          volumeMounts:
            - name: config-volume
              mountPath: /etc/logstash_c # 挂载Logstash的配置文件目录
            - name: config-yml-volume
              mountPath: /usr/share/logstash/config/ # 挂载Logstash的配置文件
            - name: timezone
              mountPath: /etc/localtime
          resources:
            limits:
              memory: 2Gi # 内存限制
              cpu: 1000m # CPU限制
            requests:
              memory: 512Mi # 内存请求
              cpu: 100m # CPU请求
      volumes:
        - name: config-volume # 卷名称
          configMap:
            name: logstash-conf # ConfigMap名称
            items:
              - key: logstash.conf # ConfigMap中的key
                path: logstash.conf # 挂载到容器中的路径
        - name: config-yml-volume # 卷名称
          configMap:
            name: logstash-yml # ConfigMap名称
            items:
              - key: logstash.yml # ConfigMap中的key
                path: logstash.yml # 挂载到容器中的路径
        - name: timezone # 卷名称
          hostPath:
            path: /etc/localtime # 宿主机路径
---
apiVersion: v1
kind: ConfigMap
metadata:
  name: logstash-conf
  namespace: kube-logging
  labels:
    type: logstash
data:
  logstash.conf: |-
    input { # 输入部分从filebeat来
      beats {
        port => 5044 # Logstash的beats端口
      }
    }
    filter { # 过滤器部分，使用Grok过滤器解析日志格式  
      if [kubernetes][container][name] == "nginx-ingress-controller" {
        json{
          source => "message" # 解析JSON格式的日志
          target => "ingress_log" # 解析后的日志存储在ingress_log字段中
        }
        if [ingress_log][requesttime] {
          mutate {
            convert => { "[ingress_log][requesttime]" => "float" } # 将requesttime转换为浮点数
          }
        }
        if [ingress_log][upstreamtime] {
          mutate {
            convert => { "[ingress_log][upstreamtime]" => "float" } # 将upstreamtime转换为浮点数
          }
        }
        if [ingress_log][status] {
          mutate {
            convert => { "[ingress_log][status]" => "float" } # 将状态码转换为整数
          }
        }
        if [ingress_log][httphost] and [ingress_log][uri] {
          mutate {
            add_field => {"[ingress_log][entry]" => "%{[ingress_log][httphost]}%{[ingress_log][uri]}"} 
          }
          mutate {
            split => { "[ingress_log][entry]" => "/" } # 将entry字段按照"/"分割
          }
          if [ingress_log][entry][1] {
            mutate {
              add_field => {"[ingress_log][entrypoint]" => "%{[ingress_log][entry][0]/%{[ingress_log][entry][1]}}"} # 添加entrypoint字段，用于记录入口点
              remove_field => "[ingress_log][entry]" # 移除原始的entry字段
            }
          }else{
            mutate {
              add_field => {"[ingress_log][entrypoint]" => "%{[ingress_log][entry][0]}/"} # 如果分割后的数组长度小于2，则直接使用原始的entry字段作为entrypoint
              remove_field => "[ingress_log][entry]" # 移除原始的entry字段
            }
          }
        }
      }
    
      # 处理以srv开头的容器日志
      if [kubernetes][container][name] =~ /^srv*/ {
        json{
          source => "message" # 解析JSON格式的日志
          target => "tmp" # tmp字段存储解析后的日志
        }
        if [kubernetes][namespace] == "kube-logging" {
          drop{} # 如果命名空间为kube-logging，则丢弃日志
        }
        if [tmp][level] {
          mutate {
            add_field => {"[applog][level]" => "%{[tmp][level]}"} # 添加level字段
          }
          if [applog][level] == "debug" {
            drop{}
          }
        }
        if [tmp][msg] {
          mutate {
            add_field => {"[applog][msg]" => "%{[tmp][msg]}"} # 添加msg字段
          }
        }
        if [tmp][func] {
          mutate {
            add_field => {"[applog][func]" => "%{[tmp][func]}"} # 添加func字段
          }
        }
        if[tmp][cost]{
          if "ms" in [tmp][cost] {
            mutate{
              split => { "[tmp][cost]" => "m" }
              add_field => {"[applog][cost]" => "%{[tmp][cost][0]}"} # 添加cost字段，并移除原始的cost字段
              convert => { "[applog][cost]" => "float" } # 将cost字段转换为浮点数
            }
          }else{
            mutate{
              add_field => {"[applog][cost]" => "%{[tmp][cost]}"} # 添加cost字段
            }
          }
        }
        if [tmp][method]{
          mutate{
            add_field => {"[applog][method]" => "%{[tmp][method]}"} # 添加method字段
          }
        }
        if [tmp][request_url]{
          mutate{
            add_field => {"[applog][request_url]" => "%{[tmp][request_url]}"} # 添加request_url字段
          }
        }
        if [tmp][meta._id]{
          mutate{
            add_field => {"[applog][mtraceId]" => "%{[tmp][meta._id]}"} # 添加meta._id字段
          }
        }
        if [tmp][project]{
          mutate{
            add_field => {"[applog][project]" => "%{[tmp][project]}"} # 添加project字段
          }
        }
        if [tmp][time]{
          mutate{
            add_field => {"[applog][time]" => "%{[tmp][time]}"} # 添加time字段
          }
        }
        if [tmp][status]{
          mutate{
            add_field => {"[applog][status]" => "%{[tmp][status]}"} # 添加status字段
            convert => { "[applog][status]" => "float" } # 将status字段转换为浮点数
          }
        }
      }
      mutate {
        remove_field => ["kubernetes", "k8s"] # 移除kubernetes和k8s字段
        remove_field => "tmp" # 移除tmp字段
        remove_field => "beat"
        remove_field => "[k8s][labels][app]" # 移除k8s.labels.app字段
      }
    }
    output {
      elasticsearch {
        hosts => ["http://elasticsearch-logging:9200"] # 输出到Elasticsearch
        index => "logstash-%{+YYYY.MM.dd}" # 设置索引名称，包含日期
        codec => "json" # 使用JSON编码
      }
    }
---
apiVersion: v1
kind: ConfigMap
metadata:
  name: logstash-yml
  namespace: kube-logging
  labels:
    type: logstash
data:
  logstash.yml: |-
    http.host: "0.0.0.0"
    xpack.monitoring.elasticsearch.hosts: http://elasticsearch-logging:9200

  4. filebeat配置文件

    yaml
    ---
apiVersion: v1
kind: ConfigMap # 配置文件，用于配置Filebeat的输入、输出和处理器
metadata:
  name: filebeat-config
  namespace: kube-logging
  labels:
    k8s-app: filebeat
data:
  filebeat.yml: |-
    filebeat.inputs:
      - type: container # 输入类型为容器
        enabled: true # 启用输入
        paths:
          - /var/log/containers/*.log # 日志路径
        processors:
          - add_kubernetes_metadata: # 添加kubernetes元数据
              host: ${NODE_NAME} # 节点名称
              matchers:
                - logs_path:
                    logs_path: "/var/log/containers/"
    output.logstash:
      hosts: ["logstash:5044"] # 输出到Logstash
      enabled: true # 启用输出
---
apiVersion: v1
kind: ServiceAccount
metadata:
  name: filebeat
  namespace: kube-logging
  labels:
    k8s-app: filebeat
---
apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRole
metadata:
  name: filebeat
  labels:
    k8s-app: filebeat
rules:
  - apiGroups: [""]
    resources: ["pods", "namespaces"]
    verbs: ["get", "list", "watch"]
---
apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRoleBinding
metadata:
  name: filebeat
  labels:
    k8s-app: filebeat
subjects:
  - kind: ServiceAccount
    name: filebeat
    namespace: kube-logging
roleRef:
  kind: ClusterRole
  name: filebeat
  apiGroup: rbac.authorization.k8s.io
---
apiVersion: apps/v1
kind: DaemonSet # 守护进程集，用于在每个节点上运行Filebeat
metadata:
  name: filebeat
  namespace: kube-logging
  labels:
    k8s-app: filebeat
spec:
  selector:
    matchLabels:
      k8s-app: filebeat
  template:
    metadata:
      labels:
        k8s-app: filebeat
    spec:
      nodeSelector:
        log: filebeat
      serviceAccountName: filebeat # 使用的服务账户
      terminationGracePeriodSeconds: 30 # 优雅终止的时间
      containers:
        - name: filebeat
          image: 192.168.31.200:5000/filebeat:7.9.3 # Filebeat镜像
          args: [
              "-c",
              "/etc/filebeat.yml", # 配置文件路径
              "-e", # 输出到标准错误
              "-httpprof",
              "0.0.0.0:6060", # 启用HTTP性能分析
            ]
          env:
            - name: NODE_NAME
              valueFrom:
                fieldRef:
                  fieldPath: spec.nodeName # 节点名称
            - name: ELASTICSEARCH_HOST
              value: elasticsearch-logging
            - name: ELASTICSEARCH_PORT
              value: "9200"
          securityContext:
            runAsUser: 0 # 以root用户运行
          resources:
            limits:
              memory: 1000Mi # 内存限制
              cpu: 1000m # CPU限制
            requests:
              memory: 100Mi # 内存请求
              cpu: 100m # CPU请求
          volumeMounts:
            - name: config
              mountPath: /etc/filebeat.yml # 挂载配置文件
              subPath: filebeat.yml
            - name: data
              mountPath: /usr/share/filebeat/data # 挂载数据目录
            - name: varlibdockercontainers
              mountPath: /var/lib
              readOnly: true
            - name: varlog
              mountPath: /var/log
              readOnly: true
            - name: timezone
              mountPath: /etc/localtime # 挂载时区文件
      volumes:
        - name: config
          configMap:
            defaultMode: 0644 # 默认权限
            name: filebeat-config # ConfigMap名称
        - name: data
          hostPath:
            path: /data/filebeat-data
            type: DirectoryOrCreate # 如果目录不存在，则创建
        - name: varlibdockercontainers
          hostPath:
            path: /var/lib
        - name: varlog
          hostPath:
            path: /var/log
        - name: inputs
          configMap:
            defaultMode: 0644 # 默认权限
            name: filebeat-inputs # ConfigMap名称
        - name: timezone
          hostPath:
            path: /etc/localtime # 宿主机路径
      tolerations:
        - effect: NoExecute
          key: dedicated
          operator: Equal
          value: gpu
        - effect: NoSchedule
          operator: Exists
